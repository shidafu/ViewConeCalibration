Get UV from XY according to 2D Homography of a projective geometry transform

Author:WANG Lei,USTB,email：towanglei@163.com

Date:2016/3/3

Algorithom:

function UV = GetUVFromXY(H,XY)

% Get UV from XY according to 2D Homography of a projective geometry transform
%
% Inputs:
%     H----3 by 3 matrix: [h11 h12 h13;
%                          h21 h22 h23;
%                          h31 h32  1]
%     XY----pointNum by cordNum matrix,
%                  cordNum==2,pointNum>=4,
%                  [x1,y1;
%                   x2,y2;
%                     :]
%
% Outputs:
%     UV----pointNum by cordNum matrix,
%                  cordNum==2,pointNum>=4,
%                  [u1,v1;
%                   u2,v2;
%                     :]

Initial

[pointNum, cordNum]=size(XY);
if ~(cordNum==2 || cordNum==3)
    error('Input matrix size error!');
end
[hH, wH]=size(H);
if hH~=3 ||~(wH==3 || wH==4)
    error('Input matrix size error!');
end
if wH==4
    H=[H(:,1:2) H(:,4)];
end
XY1=ones(pointNum,3,'double');
XY1(:,1:2)=XY(:,1:2);
UV=zeros(pointNum,2,'double');
